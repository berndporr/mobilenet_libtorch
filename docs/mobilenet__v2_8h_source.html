44 MobileNetV2(int num_classes = 1000, float width_mult = 1.0f, int round_nearest = 8, float dropout = 0.2)
45 {
46 int input_channels = 32;
47 input_channels = make_divisible(input_channels * width_mult, round_nearest);
48 features_output_channels = make_divisible(features_output_channels * std::max(1.0f, width_mult), round_nearest);
49
50 features = torch::nn::Sequential();
51
52 features->push_back(
53 Conv2dNormActivation(3,
54 input_channels,
55 /*kernel_size=*/3,
56 /*stride =*/2));
57
58 // inverted residual blocks
59 for (const auto &cfg : inverted_residual_setting)
60 {
61 const int t = cfg[0];
62 const int c = cfg[1];
63 const int n = cfg[2];
64 const int s = cfg[3];
65
66 int output_channel = make_divisible(c * width_mult, round_nearest);
67 for (int i = 0; i < n; ++i)
68 {
69 const int stride = (i == 0) ? s : 1;
70 features->push_back(
71 InvertedResidual(input_channels, output_channel, stride, t));
72 input_channels = output_channel;
73 }
74 }
75
76 features->push_back(
77 Conv2dNormActivation(input_channels,
78 features_output_channels,
79 /*kernel_size=*/1));
80
81 register_module(featuresModuleName, features);
82
83 // classifier: Dropout + Linear
84 classifier = torch::nn::Sequential();
85 classifier->push_back(torch::nn::Dropout(torch::nn::DropoutOptions(dropout)));
86 classifier->push_back(torch::nn::Linear(torch::nn::LinearOptions(features_output_channels, num_classes)));
87 register_module(classifierModuleName, classifier);
88 }

252 static torch::Tensor preprocess(cv::Mat img, bool resizeOnly = false)
253 {
254 constexpr int imageSizeBeforeCrop = 256;
255 constexpr int finalImageSize = 224;
256 constexpr int numChannels = 3; // colour
257
258 if (img.depth() != CV_8U)
259 throw std::invalid_argument("Image is not 8bit.");
260 if (img.channels() != numChannels)
261 throw std::invalid_argument("Image is not BGR / colour.");
262
263 if (resizeOnly)
264 {
265 cv::resize(img, img, cv::Size(finalImageSize, finalImageSize));
266 }
267 else
268 {
269 cv::resize(img, img, cv::Size(imageSizeBeforeCrop, imageSizeBeforeCrop));
270 constexpr int start = (imageSizeBeforeCrop - finalImageSize) / 2;
271 const cv::Rect roi(start, start, finalImageSize, finalImageSize);
272 img = img(roi).clone();
273 }
274 cv::cvtColor(img, img, cv::COLOR_BGR2RGB);
275
276 torch::Tensor tensor = torch::from_blob(img.data, {img.rows, img.cols, 3}, torch::kByte);
277 tensor = tensor.permute({2, 0, 1}).to(torch::kFloat).div_(255.0);
278 tensor = torch::data::transforms::Normalize({0.485, 0.456, 0.406}, {0.229, 0.224, 0.225})(tensor);
279 return tensor;
280 }

328 std::string ourkey2torchvision(std::string k) const
329 {
330 // called simply "conv" in the weights file
331 k = std::regex_replace(k, std::regex(InvertedResidual::className), "conv");
332 // not used at all in the weights file

338 // Makes a value divisible.
339 inline int make_divisible(int v, int divisor = 8, int min_value = -1) const
340 {
341 if (min_value < 0)
342 min_value = divisor;
343 int new_v = std::max(min_value, ((int)(((int)(v + divisor / 2)) / divisor)) * divisor);
344 if (new_v < (0.9 * (float)v))
345 new_v += divisor;
346 return new_v;
347 }

368 class Conv2dNormActivation : public torch::nn::Module
369 {
370 public:
371 static constexpr char className[] = "Conv2dNormActivation";
372
373 static inline torch::Tensor relu6(const torch::Tensor &x)
374 {
375 return torch::clamp(torch::relu(x), 0, 6);
376 }
377
378 Conv2dNormActivation(int in_channels,
379 int out_channels,
380 int kernel_size = 3,
381 int stride = 1,
382 int padding = -1,
383 int groups = 1)
384 {
385 const int dilation = 1;
386 conv = torch::nn::Sequential();
387 if (padding < 0)
388 {
389 padding = (kernel_size - 1) / 2 * dilation;
390 }
391 conv->push_back(torch::nn::Conv2d(
392 torch::nn::Conv2dOptions(in_channels, out_channels, kernel_size)
393 .stride(stride)
394 .padding(padding)
395 .groups(groups)
396 .bias(false)));
397 conv->push_back(torch::nn::BatchNorm2d(torch::nn::BatchNorm2dOptions(out_channels)));
398 conv->push_back(torch::nn::Functional(relu6));
399 register_module(className, conv);
400 }

402 torch::Tensor forward(const torch::Tensor &x)
403 {
404 return conv->forward(x);

415 class InvertedResidual : public torch::nn::Module
416 {
417 public:
418 static constexpr char className[] = "InvertedResidual";
419
420 InvertedResidual(int inp, int oup, int stride, int expand_ratio)
421 {
422 if ((stride < 1) || (stride > 2))
423 {
424 throw std::invalid_argument("Stride needs to be 1 or 2.");
425 }
426 const int hidden_dim = (int)round(inp * expand_ratio);
427 use_res_connect = (stride == 1) && (inp == oup);
428
429 conv = torch::nn::Sequential();
430
431 if (expand_ratio != 1)
432 {
433 conv->push_back(
434 Conv2dNormActivation(inp,
435 hidden_dim,
436 /*kernel_size*/ 1));
437 }
438
439 conv->push_back(
440 Conv2dNormActivation(hidden_dim,
441 hidden_dim,
442 /*kernel_size=*/3,
443 /*stride=*/stride,
444 /*padding=*/-1,
445 /*groups=*/hidden_dim));
446
447 conv->push_back(torch::nn::Conv2d(
448 torch::nn::Conv2dOptions(hidden_dim, oup,
449 /*kernel_size=*/1)
450 .stride(1)
451 .padding(0)
452 .bias(false)));
453 conv->push_back(torch::nn::BatchNorm2d(torch::nn::BatchNorm2dOptions(oup)));
454
455 register_module(className, conv);
456 }
457
458 torch::Tensor forward(const torch::Tensor &x)
459 {
460 if (use_res_connect)
461 {
462 return x + conv->forward(x);
463 }
464 else
465 {
466 return conv->forward(x);
467 }
468 }
469 torch::nn::Sequential conv{nullptr};
470 bool use_res_connect;
471 };
