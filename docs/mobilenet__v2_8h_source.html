30class Conv2dNormActivation : public torch::nn::Module
31{
32public:
33 static constexpr char className[] = "Conv2dNormActivation";
34
35 static inline torch::Tensor relu6(const torch::Tensor &x)
36 {
37 return torch::clamp(torch::relu(x), 0, 6);
38 }
39
40 Conv2dNormActivation(int in_channels,
41 int out_channels,
42 int kernel_size = 3,
43 int stride = 1,
44 int padding = -1,
45 int groups = 1)
46 {
47 const int dilation = 1;
48 conv = torch::nn::Sequential();
49 if (padding < 0)
50 {
51 padding = (kernel_size - 1) / 2 * dilation;
52 }
53 conv->push_back(torch::nn::Conv2d(
54 torch::nn::Conv2dOptions(in_channels, out_channels, kernel_size)
55 .stride(stride)
56 .padding(padding)
57 .groups(groups)
58 .bias(false)));
59 conv->push_back(torch::nn::BatchNorm2d(torch::nn::BatchNorm2dOptions(out_channels)));
60 conv->push_back(torch::nn::Functional(relu6));
61 register_module(className, conv);
62 }
63
64 torch::Tensor forward(const torch::Tensor &x)
65 {
66 return conv->forward(x);
67 }
68
69private:
70 torch::nn::Sequential conv{nullptr};
71};

77class InvertedResidual : public torch::nn::Module
78{
79public:
80 static constexpr char className[] = "InvertedResidual";
81
82 InvertedResidual(int inp, int oup, int stride, int expand_ratio)
83 {
84 if ((stride < 1) || (stride > 2))
85 {
86 throw std::invalid_argument("Stride needs to be 1 or 2.");
87 }
88 const int hidden_dim = (int)round(inp * expand_ratio);
89 use_res_connect = (stride == 1) && (inp == oup);
90
91 conv = torch::nn::Sequential();
92
93 if (expand_ratio != 1)
94 {
95 conv->push_back(
96 Conv2dNormActivation(inp,
97 hidden_dim,
98 /*kernel_size*/ 1));
99 }
100
101 conv->push_back(
102 Conv2dNormActivation(hidden_dim,
103 hidden_dim,
104 /*kernel_size=*/3,
105 /*stride=*/stride,
106 /*padding=*/-1,
107 /*groups=*/hidden_dim));
108
109 conv->push_back(torch::nn::Conv2d(
110 torch::nn::Conv2dOptions(hidden_dim, oup,
111 /*kernel_size=*/1)
112 .stride(1)
113 .padding(0)
114 .bias(false)));
115 conv->push_back(torch::nn::BatchNorm2d(torch::nn::BatchNorm2dOptions(oup)));
116
117 register_module(className, conv);
118 }
119
120 torch::Tensor forward(const torch::Tensor &x)
121 {
122 if (use_res_connect)
123 {
124 return x + conv->forward(x);
125 }
126 else
127 {
128 return conv->forward(x);
129 }
130 }
131 torch::nn::Sequential conv{nullptr};
132 bool use_res_connect;
133};

141public:
142 // Module name of the feature detector.

162 MobileNetV2(int num_classes = 1000, float width_mult = 1.0f, int round_nearest = 8, float dropout = 0.2)
163 {
164 int input_channels = 32;
165 input_channels = make_divisible(input_channels * width_mult, round_nearest);
166 features_output_channels = make_divisible(features_output_channels * std::max(1.0f, width_mult), round_nearest);
167
168 features = torch::nn::Sequential();
169
170 features->push_back(
171 Conv2dNormActivation(3,
172 input_channels,
173 /*kernel_size=*/3,
174 /*stride =*/2));
175
176 // inverted residual blocks
177 for (const auto &cfg : inverted_residual_setting)
178 {
179 const int t = cfg[0];
180 const int c = cfg[1];
181 const int n = cfg[2];
182 const int s = cfg[3];
183
184 int output_channel = make_divisible(c * width_mult, round_nearest);
185 for (int i = 0; i < n; ++i)
186 {
187 const int stride = (i == 0) ? s : 1;
188 features->push_back(
189 InvertedResidual(input_channels, output_channel, stride, t));
190 input_channels = output_channel;
191 }
192 }
193
194 features->push_back(
195 Conv2dNormActivation(input_channels,
196 features_output_channels,
197 /*kernel_size=*/1));
198
199 register_module(featuresModuleName, features);
200
201 // classifier: Dropout + Linear
202 classifier = torch::nn::Sequential();
203 classifier->push_back(torch::nn::Dropout(torch::nn::DropoutOptions(dropout)));
204 classifier->push_back(torch::nn::Linear(torch::nn::LinearOptions(features_output_channels, num_classes)));
205 register_module(classifierModuleName, classifier);
206 }
207
215 int getNinputChannels4Classifier() const
216 {
217 return features_output_channels;
218 }
219
226 torch::Tensor forward(torch::Tensor x)
227 {
228 x = features->forward(x);
229 const torch::nn::functional::AdaptiveAvgPool2dFuncOptions &ar = torch::nn::functional::AdaptiveAvgPool2dFuncOptions({1, 1});
230 x = torch::nn::functional::adaptive_avg_pool2d(x, ar);
231 x = torch::flatten(x, 1);
232 x = classifier->forward(x);
233 return x;
234 }

239 void initialize_weights()
240 {
241 for (auto &module : modules(/*include_self=*/false))
242 {
243 if (auto M = dynamic_cast<torch::nn::Conv2dImpl *>(module.get()))
244 {
245 torch::nn::init::kaiming_normal_(M->weight, /*a=*/0, torch::kFanOut, torch::kReLU);
246 if (M->options.bias())
247 torch::nn::init::zeros_(M->bias);
248 }
249 else if (auto M = dynamic_cast<torch::nn::BatchNorm2dImpl *>(module.get()))
250 {
251 torch::nn::init::ones_(M->weight);
252 torch::nn::init::zeros_(M->bias);
253 }
254 else if (auto M = dynamic_cast<torch::nn::LinearImpl *>(module.get()))
255 {
256 torch::nn::init::normal_(M->weight, 0.0, 0.01);
257 torch::nn::init::zeros_(M->bias);
258 }
259 }

356 static torch::Tensor preprocess(cv::Mat img, bool resizeOnly = false)
357 {
358 constexpr int imageSizeBeforeCrop = 256;
359 constexpr int finalImageSize = 224;
360 constexpr int numChannels = 3; // colour
361
362 if (img.depth() != CV_8U)
363 throw std::invalid_argument("Image is not 8bit.");
364 if (img.channels() != numChannels)
365 throw std::invalid_argument("Image is not BGR / colour.");
366
367 if (resizeOnly)
368 {
369 cv::resize(img, img, cv::Size(finalImageSize, finalImageSize));
370 }
371 else
372 {
373 cv::resize(img, img, cv::Size(imageSizeBeforeCrop, imageSizeBeforeCrop));
374 constexpr int start = (imageSizeBeforeCrop - finalImageSize) / 2;
375 const cv::Rect roi(start, start, finalImageSize, finalImageSize);
376 img = img(roi).clone();
377 }
378 cv::cvtColor(img, img, cv::COLOR_BGR2RGB);
379
380 torch::Tensor tensor = torch::from_blob(img.data, {img.rows, img.cols, 3}, torch::kByte);
381 tensor = tensor.permute({2, 0, 1}).to(torch::kFloat).div_(255.0);
382 tensor = torch::data::transforms::Normalize({0.485, 0.456, 0.406}, {0.229, 0.224, 0.225})(tensor);
383 return tensor;
384 }

411 // makes a value divisible
412 inline int make_divisible(int v, int divisor = 8, int min_value = -1) const
413 {
414 if (min_value < 0)
415 min_value = divisor;
416 int new_v = std::max(min_value, ((int)(((int)(v + divisor / 2)) / divisor)) * divisor);
417 if (new_v < (0.9 * (float)v))
418 new_v += divisor;
419 return new_v;
420 }
